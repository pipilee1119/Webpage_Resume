使用python實作BST。

程式

                        class Node:
    def __init__ (self,val = None, left = None , right = None):
        self.val = val
        self.left = left
        self.right = right

class BST:
    
    def __init__(self):
        self.root = None
        
        
    def _min_node(self,node):
        if node.left:
            self.min_node(node.left)
        else:
            return node
        
    def insert(self,val):
        if not self.root:
            self.root = Node(val)
        else:
            self._insert(val,self.root)
            
    def _insert(self,val,node):
        if not node:
            return
        if val > node.val:
            if not node.right:
                node.right = Node(val)
            else:
                self._insert(val,node.right)
        elif val < node.val:
            if not node.left:
                node.left = Node(val)
            else:
                self._insert(val,node.left)

        
            
    def remove(self,val):
        self.root = self._remove(val,self.root)
        
    def _remove(self,val,node):
        if not node:
            print('沒有該數字')
            return None
        if val > node.val:
            node.right = self._remove(val,node.right)
        elif val < node.val:
            node.left = self._remove(val,node.left)
        else:
            if not node.right and not node.left:
                return None
            elif not node.right:
                return node.left
            elif not node.left:
                return node.right
            else:
                tmp = self._min_node(node.right)
                node.right = self._remove(tmp.val,node.right)
                tmp.left = node.left
                tmp.right = node.right
                return tmp
        return node
        
        
    def print_node(self):
        if self.root:
            self._print_node(self.root)
    
    def _print_node(self,node):
        if node:
            self._print_node(node.left)
            print(node.val)
            self._print_node(node.right)
